# Variant 5 Grammar
# Lines starting with '#' are comments and are ignored.
# Blank lines are ignored.
#
# Format:
#   NONTERMINALS: comma-separated list
#   TERMINALS:    comma-separated list
#   START:        single symbol
#   PRODUCTIONS:  one per line,  LHS -> sym1 sym2 ...
#                 use 'eps' for an epsilon production

NONTERMINALS: S, A, B, C, D
TERMINALS: a, b, d
START: S

PRODUCTIONS:
S -> d B
S -> A
A -> d
A -> d S
A -> a B d B
B -> a
B -> a S
B -> A C
D -> A B
C -> b C
C -> eps
